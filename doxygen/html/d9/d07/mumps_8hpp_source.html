188 IT solve(sparse_coo_mat<DT, IT>&& A, full_mat<DT, IT>&& B) {
189 if(A.n != B.n_rows) return IT{-1};

191 // ReSharper disable CppCStyleCast

193 id.nnz = A.nnz;
194 id.irn = A.row;
195 id.jcn = A.col;
196 id.a = (entry_t*)A.data;
197
198 id.lrhs = B.n_rows;
199 id.nrhs = B.n_cols;
200 id.rhs = (entry_t*)B.data;

203 perform_job(6);
