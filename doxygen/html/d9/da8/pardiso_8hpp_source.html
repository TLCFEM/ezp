104 IT solve(sparse_csr_mat<DT, IT>&& A, full_mat<DT, IT>&& B) {
105 if(A.n != B.n_rows) return -1;
106
107 std::vector<DT> b_ref;
108 if(0 == comm_world.rank()) b_ref.resize(B.n_rows * B.n_cols);
109
110 void *b_ptr, *x_ptr;
111
112 if(0 == iparm[0] || 0 == iparm[5]) {
113 // b unchanged, x has solution
114 if(0 == comm_world.rank()) std::copy(B.data, B.data + b_ref.size(), b_ref.data());
115 b_ptr = b_ref.data();
116 x_ptr = B.data;
117 }
118 else {
119 // b has solution, x is still used
120 b_ptr = B.data;
121 x_ptr = b_ref.data();
122 }
123
124 IT error = -1;
125
126 IT phase = 13;
127 if constexpr(sizeof(IT) == 4) {
128 using E = std::int32_t;
129 cluster_sparse_solver(pt, (E*)&one, (E*)&one, (E*)&mtype, (E*)&phase, (E*)&A.n, A.data, (E*)A.irn, (E*)A.jcn, nullptr, (E*)&B.n_cols, (E*)iparm, (E*)&msglvl, b_ptr, x_ptr, &comm, (E*)&error);
130 }
131 else if constexpr(sizeof(IT) == 8) {
132 using E = std::int64_t;
133 cluster_sparse_solver_64(pt, (E*)&one, (E*)&one, (E*)&mtype, (E*)&phase, (E*)&A.n, A.data, (E*)A.irn, (E*)A.jcn, nullptr, (E*)&B.n_cols, (E*)iparm, (E*)&msglvl, b_ptr, x_ptr, &comm, (E*)&error);
134 }
135
136 phase = -1;
137 if constexpr(sizeof(IT) == 4) {
138 using E = std::int32_t;
139 cluster_sparse_solver(pt, (E*)&one, (E*)&one, (E*)&mtype, (E*)&phase, (E*)&A.n, nullptr, (E*)A.irn, (E*)A.jcn, nullptr, (E*)&B.n_cols, (E*)iparm, (E*)&msglvl, nullptr, nullptr, &comm, (E*)&error);
140 }
