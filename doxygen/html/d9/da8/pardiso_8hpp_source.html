109 IT solve(sparse_csr_mat<DT, IT>&& A, full_mat<DT, IT>&& B) {
110 IT error = 0;
111 if(0 == comm_world.rank() && A.row_ptr[A.n] != A.nnz + (0 == iparm[0] || 0 == iparm[34])) error = -1;
112
113 error = sync_error(error);
114 if(error < 0) return error;
115
116 if(A.n != B.n_rows) return -1;
117
118 iparm[39] = 0; // force centralised input/output
119
120 if constexpr(std::is_same_v<DT, double> || std::is_same_v<DT, complex16>) iparm[27] = 0;
121 else if constexpr(std::is_same_v<DT, float> || std::is_same_v<DT, complex8>) iparm[27] = 1;
122
123 std::vector<DT> b_ref;
124 if(0 == comm_world.rank()) b_ref.resize(B.n_rows * B.n_cols);
125
126 void *b_ptr, *x_ptr;
127
128 if(0 == iparm[0] || 0 == iparm[5]) {
129 // b unchanged, x has solution
130 if(0 == comm_world.rank()) std::copy_n(B.data, b_ref.size(), b_ref.data());
131 b_ptr = b_ref.data();
132 x_ptr = B.data;
133 }
134 else {
135 // b has solution, x is still used
136 b_ptr = B.data;
137 x_ptr = b_ref.data();
138 }
139
140 IT phase = 13;
141 // ReSharper disable CppCStyleCast
142 if constexpr(sizeof(IT) == 4) {
143 using E = std::int32_t;
144 cluster_sparse_solver(pt, (E*)&one, (E*)&one, (E*)&mtype, (E*)&phase, (E*)&A.n, A.data, (E*)A.row_ptr, (E*)A.col_idx, nullptr, (E*)&B.n_cols, (E*)iparm, (E*)&msglvl, b_ptr, x_ptr, &comm, (E*)&error);
145 }

149 }

154 phase = -1;
