85 auto init_storage(const IT n, const IT kl, const IT ku) {

117 auto operator()(const IT i, const IT j) const {
118 if(i < 0 || i >= n || j < 0 || j >= n) return IT{-1};
119 if(i - j > kl || j - i > ku) return IT{-1};
120 return ku + i + j * (kl + ku);
121 }
