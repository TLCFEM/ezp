138 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
139 // ReSharper disable CppCStyleCast
140 if(std::is_same_v<DT, double>) {
141 using E = double;
142 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
143 }
144 else {
145 using E = float;
146 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
147 }
148 // ReSharper restore CppCStyleCast
149 }
150
151 public:
152 IT n_rows, n_cols, context{-1}, rank{-1}, size{-1}, my_row{-1}, my_col{-1};
153
154 blacs_context()
155 : blacs_context(get_env<IT>().size(), 1) {}
156
157 blacs_context(const char order)
158 : n_rows(-1)
159 , n_cols(-1) {
160 const auto& env = get_env<IT>();
161 n_rows = std::max(1, static_cast<IT>(std::sqrt(env.size())));
162 n_cols = env.size() / n_rows;
163 init(order);
164 };
