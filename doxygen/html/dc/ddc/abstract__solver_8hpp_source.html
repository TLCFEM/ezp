138 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
139 // ReSharper disable CppCStyleCast
140 if(std::is_same_v<DT, double>) {
141 using E = double;
142 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
143 }
144 else if (std::is_same_v<DT, float>) {
145 using E = float;
146 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
147 }else if(std::is_same_v<DT, complex16>) {
148 using E = complex16;
149 pzgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
150 }
151 else if (std::is_same_v<DT, complex8>) {
152 using E = complex8;
153 pcgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
154 }
155 // ReSharper restore CppCStyleCast
156 }

228 template<data_t DT> auto scatter(const full_mat<DT, IT>& A, const desc<IT>& desc_a, std::vector<DT>& B, const desc<IT>& desc_b) {
229 if(!A.distributed) return copy_to(A.data, desc_a.data(), B.data(), desc_b.data());
