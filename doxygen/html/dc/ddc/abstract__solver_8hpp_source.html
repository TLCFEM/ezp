161 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
162 // ReSharper disable CppCStyleCast
163 if(std::is_same_v<DT, double>) {
164 using E = double;
165 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
166 }
167 else if(std::is_same_v<DT, float>) {
168 using E = float;
169 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
170 }
171 else if(std::is_same_v<DT, complex16>) {
172 using E = complex16;
173 pzgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
174 }
175 else if(std::is_same_v<DT, complex8>) {
176 using E = complex8;
177 pcgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
178 }
179 // ReSharper restore CppCStyleCast
180 }

183 IT n_rows, n_cols, context{-1}, rank{-1}, size{-1}, my_row{-1}, my_col{-1};

252 template<data_t DT> auto scatter(const full_mat<DT, IT>& A, const desc<IT>& desc_a, std::vector<DT>& B, const desc<IT>& desc_b) {
253 if(!A.distributed) return copy_to(A.data, desc_a.data(), B.data(), desc_b.data());
