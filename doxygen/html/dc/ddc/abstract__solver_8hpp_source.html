192 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
193 // ReSharper disable CppCStyleCast
194 if(std::is_same_v<DT, double>) {
195 using E = double;
196 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
197 }
198 else if(std::is_same_v<DT, float>) {
199 using E = float;
200 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
201 }
202 else if(std::is_same_v<DT, complex16>) {
203 using E = complex16;
204 pzgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
205 }
206 else if(std::is_same_v<DT, complex8>) {
207 using E = complex8;
208 pcgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
209 }
210 // ReSharper restore CppCStyleCast
211 }

214 IT n_rows, n_cols, context{-1}, rank{-1}, size{-1}, my_row{-1}, my_col{-1};

375 template<full_container_t CT> auto to_full(CT&& custom) {
376 if constexpr(has_mem<CT>) return full_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.mem()};
377 else if constexpr(has_memptr<CT>) return full_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.memptr()};
378 else if constexpr(has_data_method<CT>) return full_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.data()};
379 else if constexpr(has_iterator<CT>) return full_mat<DT, IT>{custom.n_rows, custom.n_cols, &(*custom.begin())};
380 else static_assert(always_false_v<CT>, "invalid container type");
381 }
382
383 template<band_container_t CT> auto to_band(CT&& custom) {
384 if constexpr(has_mem<CT>) return band_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.kl, custom.ku, custom.mem()};
385 else if constexpr(has_memptr<CT>) return band_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.kl, custom.ku, custom.memptr()};
386 else if constexpr(has_data_method<CT>) return band_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.kl, custom.ku, custom.data()};
387 else if constexpr(has_iterator<CT>) return band_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.kl, custom.ku, &(*custom.begin())};
388 else static_assert(always_false_v<CT>, "invalid container type");
389 }
390
391 template<band_symm_container_t CT> auto to_band_symm(CT&& custom) {
392 if constexpr(has_mem<CT>) return band_symm_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.klu, custom.mem()};
393 else if constexpr(has_memptr<CT>) return band_symm_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.klu, custom.memptr()};
394 else if constexpr(has_data_method<CT>) return band_symm_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.klu, custom.data()};
395 else if constexpr(has_iterator<CT>) return band_symm_mat<DT, IT>{custom.n_rows, custom.n_cols, custom.klu, &(*custom.begin())};
396 else static_assert(always_false_v<CT>, "invalid container type");
397 }
