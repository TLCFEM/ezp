129 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
130 // ReSharper disable CppCStyleCast
131 if(std::is_same_v<DT, double>) {
132 using E = double;
133 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
134 }
135 else {
136 using E = float;
137 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
138 }
139 // ReSharper restore CppCStyleCast
140 }

142 public:
143 IT n_rows, n_cols, context{-1}, rank{-1}, size{-1}, my_row{-1}, my_col{-1};

145 blacs_context()
146 : blacs_context(get_env<IT>().size(), 1) {}

148 blacs_context(const char order)
149 : n_rows(-1)
150 , n_cols(-1) {
151 const auto& env = get_env<IT>();
152 n_rows = std::max(1, static_cast<IT>(std::sqrt(env.size())));
153 n_cols = env.size() / n_rows;
154 init(order);
155 };
