243 template<data_t DT> auto copy_to(const DT* A, const IT* desc_a, DT* B, const IT* desc_b) {
244 // ReSharper disable CppCStyleCast
245 if(std::is_same_v<DT, double>) {
246 using E = double;
247 pdgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
248 }
249 else if(std::is_same_v<DT, float>) {
250 using E = float;
251 psgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
252 }
253 else if(std::is_same_v<DT, complex16>) {
254 using E = complex16;
255 pzgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
256 }
257 else if(std::is_same_v<DT, complex8>) {
258 using E = complex8;
259 pcgemr2d(desc_a + 2, desc_a + 3, (E*)A, &ONE, &ONE, desc_a, (E*)B, &ONE, &ONE, desc_b, &context);
260 }
261 // ReSharper restore CppCStyleCast
262 }

264 public:
265 IT n_rows, n_cols, context{-1}, rank{-1}, size{-1}, my_row{-1}, my_col{-1};
